class Solution {
    public double minimumAverage(int[] nums) {
        int n = nums.length;
        double result = 51.0;

        Arrays.sort(nums);

        for(int i=0; i<n/2; i++) {
            double avg = (nums[i] + nums[n-i-1]) / 2.0;
            result = Math.min(result, avg);
        }

        return result;
    }
}
