class Solution {
    public long sumAndMultiply(int n) {
        long val = 0, sum = 0, mul = 1;

        while(n > 0) {
            int d = n % 10;
            if(d != 0) {
                sum += d;
                val = val + mul * d;
                mul *= 10;
            }
            n /= 10;
        }

        return val * sum;
    }
}
