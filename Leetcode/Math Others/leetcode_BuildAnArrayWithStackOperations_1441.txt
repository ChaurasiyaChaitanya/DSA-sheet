class Solution {
    public List<String> buildArray(int[] target, int n) {
        List<String> result = new ArrayList<>();
        int curr = 1;

        for(int num : target) {
            while(curr < num) {
                result.add("Push");
                result.add("Pop");
                curr++;
            }
            result.add("Push");
            curr++;
        }

        return result;
    }
}
