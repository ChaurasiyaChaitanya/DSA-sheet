class Solution {
    public int secondHighest(String s) {
        int max1 = -1, max2 = -1;

        for(char c : s.toCharArray()) {
            if(Character.isDigit(c)) {
                int d = c - '0';
                if(d > max1) {
                    max2 = max1;
                    max1 = d;
                } else if(d > max2 && d != max1) {
                    max2 = d;
                }
            }
        }

        return max2;
    }
}
