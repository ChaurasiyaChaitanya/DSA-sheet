class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        Map<Integer, Integer> map = new HashMap<>();
        int result = 0;

        for(int i=lowLimit; i<=highLimit; i++) {
            int num = i, sum = 0;

            while(num > 0) {
                sum += num % 10;
                num /= 10;
            }

            map.put(sum, map.getOrDefault(sum, 0) + 1);
            result = Math.max(result, map.get(sum));
        }

        return result;
    }
}
