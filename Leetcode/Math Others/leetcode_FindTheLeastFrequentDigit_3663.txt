class Solution {
    public int getLeastFrequentDigit(int n) {
        int[] freq = new int[10];

        while(n > 0) {
            freq[n % 10]++;
            n /= 10;
        }

        int count = Integer.MAX_VALUE, result = 0;

        for(int i=0; i<10; i++) {
            if(freq[i] > 0 && freq[i] < count) {
                result = i;
                count = freq[i];
            }
        }

        return result;
    }
}
