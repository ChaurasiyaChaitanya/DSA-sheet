class Solution {
    public int countCompleteDayPairs(int[] hours) {
        Map<Integer, Integer> map = new HashMap<>();
        int result = 0;

        for(int hour : hours) {
            int rem = hour % 24;
            int comp = (24 - rem) % 24;

            result += map.getOrDefault(comp, 0);

            map.put(rem, map.getOrDefault(rem, 0) + 1);
        }

        return result;
    }
}
