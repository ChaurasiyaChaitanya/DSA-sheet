class Solution {
    public int countDigitOccurrences(int[] nums, int digit) {
        int result = 0;

        for(int num : nums) {
            if(num == 0 && digit == 0) {
                result++;
            }

            while(num > 0) {
                if(num % 10 == digit) {
                    result++;
                }
                num /= 10;
            }
        }

        return result;
    }
}
