class Solution {
    public int titleToNumber(String columnTitle) {
        int result = 0;

        // "AA" → 1×26 + 1 = 27
        // "AB" → 1×26 + 2 = 28

        for(char c : columnTitle.toCharArray()) {
            int val = c - 'A' + 1; //  as A = 1
            result = result * 26 + val;
        }

        return result;
    }
}
