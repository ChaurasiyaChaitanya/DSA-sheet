class Solution {
    public long removeZeros(long n) {
        long result = 0;
        long mul = 1;

        while(n > 0) {
            long d = n % 10;
            if(d != 0) {
                result += d * mul;
                mul *= 10;
            }
            n /= 10;
        }

        return result;
    }
}
