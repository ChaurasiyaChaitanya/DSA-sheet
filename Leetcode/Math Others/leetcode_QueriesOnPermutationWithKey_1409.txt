class Solution {
    public int[] processQueries(int[] queries, int m) {
        int n = queries.length;
        List<Integer> list = new ArrayList<>(m);

        for(int i=1; i<=m; i++) {
            list.add(i);
        }

        int[] result = new int[n];

        for(int i=0; i<n; i++) {
            result[i] = list.indexOf(queries[i]);
            list.remove(result[i]);
            list.add(0, queries[i]);
        }

        return result;
    }
}
