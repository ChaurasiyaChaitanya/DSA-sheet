class Solution {
    public int bulbSwitch(int n) {
        // An odd number of toggles means it ends in the "on" state
        // Perfect squares have an odd number of divisors
        // All the perfect squares in the range of 1 to n toggle odd times

        // For example, the divisors of 9 (a perfect square) are 1, 3, 9 (3 divisors - odd).
        // The divisors of 10 (not a perfect square) are 1, 2, 5, 10 (4 divisors - even).

        return (int) Math.sqrt(n);
    }
}
