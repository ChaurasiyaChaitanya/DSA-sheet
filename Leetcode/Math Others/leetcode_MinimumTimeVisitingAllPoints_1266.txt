class Solution {
    public int minTimeToVisitAllPoints(int[][] points) {
        int result = 0;

        for(int i=1; i<points.length; i++) {
            int x1 = points[i-1][0], y1 = points[i-1][1];
            int x2 = points[i][0], y2 = points[i][1];

            result += Math.max(Math.abs(x2 - x1), Math.abs(y2 - y1));
        }

        return result;
    }
}
