class Solution {
    private static final int MOD = 1_000_000_007;

    public int numSub(String s) {
        int count = 0;
        int result = 0;

        for(char c : s.toCharArray()) {
            if(c == '1') {
                count++;
                result = (result + count) % MOD;
            } else {
                count = 0;
            }
        }

        return result;
    }
}
