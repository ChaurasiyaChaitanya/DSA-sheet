class Solution {
    int n;

    public boolean canReach(int[] arr, int start) {
        n = arr.length;
        return dfs(arr, start);    
    }

    private boolean dfs(int[] arr, int i) {
        if(i < 0 || i >= n || arr[i] < 0) {
            return false;
        }
        if(arr[i] == 0) {
            return true;
        }

        arr[i] *= -1;

        return dfs(arr, i - arr[i]) || dfs(arr, i + arr[i]);
    }
}
