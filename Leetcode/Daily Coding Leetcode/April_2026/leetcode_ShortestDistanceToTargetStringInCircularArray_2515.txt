class Solution {
    public int closestTarget(String[] words, String target, int startIndex) {
        int n = words.length;
        int result = n;

        for(int i=0; i<n; i++) {
            if(words[i].equals(target)) {
                int d = Math.abs(i - startIndex);
                result = Math.min(result, Math.min(d, n - d));
            }
        }

        return result == n ? -1 : result;
    }
}
