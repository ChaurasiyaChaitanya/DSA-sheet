class Solution {
    private static final int MOD = 1_000_000_007;

    public int peopleAwareOfSecret(int n, int delay, int forget) {
        // dp[day] = number of people who first learn the secret on "day"
        int[] dp = new int[n+1];

        dp[1] = 1;
        long count = 0; // number of people who can share the secret today

        for (int day=2; day<=n; day++) {
            // people who become eligible to share on this day
            if (day - delay > 0) {
                count = (count + dp[day-delay]) % MOD;
            }
            // people who forget on this day
            if (day - forget > 0) {
                count = (count - dp[day-forget] + MOD) % MOD;
            }
            dp[day] = (int) count; // total number of people who will get to know today
        }

        // count the people who still remember on day n
        int result = 0;
        for (int day=n-forget+1; day<=n; day++) {
            if (day > 0) {
                result = (result + dp[day]) % MOD;
            }
        }

        return result;
    }
}
