class Solution {
    public int makeTheIntegerZero(int num1, int num2) {
        for(int i=1; i<=36; i++) {
            long num = (long) num1 - (long) i * num2;

            if(num < 0) 
                return -1;

            if (Long.bitCount(num) <= i && i <= num) {
                return i;
            }
        }

        return -1;
    }
}
