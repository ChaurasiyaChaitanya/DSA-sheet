class Solution {
    public long maxMatrixSum(int[][] matrix) {
        int count = 0;
        long sum = 0;
        long minVal = Long.MAX_VALUE;

        for(int[] row : matrix) {
            for(int val : row) {
                if(val < 0) count++;

                sum += Math.abs(val);
                minVal = Math.min(minVal, Math.abs(val));
            }
        }

        return count % 2 == 0 ? sum : sum - 2 * minVal;
    }
}
