class Solution {
    public int minPairSum(int[] nums) {
        Arrays.sort(nums);

        int n = nums.length;
        int result = 0;

        for(int i=0; i<n/2; i++) {
            result = Math.max(result, nums[i] + nums[n-i-1]);
        }

        // int i = 0, j = n - 1;

        // while(i < j) {
        //     result = Math.max(result, nums[i] + nums[j]);
        //     i++; j--;
        // }

        return result;
    }
}
