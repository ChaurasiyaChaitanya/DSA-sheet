/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    private static final int MOD = 1_000_000_007;
    long totalSum = 0;
    long result = 0;

    public int maxProduct(TreeNode root) {
        totalSum = findTotalSum(root);

        dfs(root);

        return (int) (result % MOD);
    }

    private long findTotalSum(TreeNode root) {
        if(root == null) {
            return 0;
        }

        return root.val + findTotalSum(root.left) + findTotalSum(root.right);
    }

    private long dfs(TreeNode root) {
        if(root == null) {
            return 0;
        }

        long leftSum = dfs(root.left);
        long rightSum = dfs(root.right);

        long currSum = root.val + leftSum + rightSum;

        long prod = currSum * (totalSum - currSum);
        result = Math.max(result, prod);

        return currSum;
    }
}
