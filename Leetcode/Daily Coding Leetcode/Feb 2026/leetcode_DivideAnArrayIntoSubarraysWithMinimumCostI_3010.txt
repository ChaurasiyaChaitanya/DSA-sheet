class Solution {
    public int minimumCost(int[] nums) {
        int minEle = Integer.MAX_VALUE, secondMinEle = Integer.MAX_VALUE;

        for(int i=1; i<nums.length; i++) {
            if(nums[i] < minEle) {
                secondMinEle = minEle;
                minEle = nums[i];
            } else if(nums[i] < secondMinEle) {
                secondMinEle = nums[i];
            }
        }

        return nums[0] + minEle + secondMinEle;
    }
}
