class Solution {
    private static final int MOD = 1_000_000_007;

    public int concatenatedBinary(int n) {
        long result = 0;
        int bits = 0;

        for(int i=1; i<=n; i++) {
            if((i & (i - 1)) == 0)  
                bits++;
            
            result = ((result << bits) % MOD + i) % MOD;
        }

        return (int) result;
    }
}
