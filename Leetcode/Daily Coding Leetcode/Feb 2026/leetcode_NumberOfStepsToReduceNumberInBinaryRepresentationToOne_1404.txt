class Solution {
    public int numSteps(String s) {
        int n = s.length();
        int carry = 0, result = 0;

        for(int i=n-1; i>=1; i--) {
            if(((s.charAt(i) - '0') + carry) % 2 == 1) {
                result += 2;
                carry = 1;
            } else {
                result += 1;
            }
        }

        return result + carry;
    }
}
