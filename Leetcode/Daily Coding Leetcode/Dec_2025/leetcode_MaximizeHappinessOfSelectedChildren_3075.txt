class Solution {
    public long maximumHappinessSum(int[] happiness, int k) {
        Arrays.sort(happiness);

        int n = happiness.length - 1;
        long result = 0;

        for(int i=0; i<k; i++) {
            int val = happiness[n-i] - i;
            if(val > 0) {
                result += val;
            } else {
                break;
            }
        }

        return result;
    }
}
