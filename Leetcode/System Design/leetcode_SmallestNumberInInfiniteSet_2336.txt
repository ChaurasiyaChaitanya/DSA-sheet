class SmallestInfiniteSet {
    Set<Integer> set;
    int curr;

    public SmallestInfiniteSet() {
        set = new HashSet<>();
        curr = 1;
    }
    
    public int popSmallest() {
        if(!set.isEmpty()) {
            int smallest = Collections.min(set);
            set.remove(smallest);
            return smallest;
        } else {
            return curr++;
        }
    }
    
    public void addBack(int num) {
        if(curr > num) {
            set.add(num);
        }
    }
}

 * int param_1 = obj.popSmallest();
