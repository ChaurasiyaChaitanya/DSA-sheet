class Solution {
    public int countKDifference(int[] nums, int k) {
        int[] freq = new int[101];

        for(int num : nums) {
            freq[num]++;
        }

        int result = 0;
        for(int num : nums) {
            if(num - k > 0 && freq[num - k] > 0) {
                result += freq[num - k];
            }
        }

        return result;
    }
}
