class Solution {
    public int sumDivisibleByK(int[] nums, int k) {
        int[] freq = new int[101];

        for(int num : nums) {
            freq[num]++;
        }

        int result = 0;

        for(int i=1; i<freq.length; i++) {
            if(freq[i] != 0 && freq[i] % k == 0) {
                result += freq[i] * i;
            }
        }

        return result;
    }
}
