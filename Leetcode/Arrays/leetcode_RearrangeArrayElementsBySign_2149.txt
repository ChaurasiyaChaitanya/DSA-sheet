class Solution {
    public int[] rearrangeArray(int[] nums) {
        int n = nums.length;
        int i = 0, j = 1; // i -> pos idx, j -> neg idx

        int[] result = new int[n];

        for(int num : nums) {
            if(num > 0) {
                result[i] = num;
                i+=2;
            } else {
                result[j] = num;
                j+=2;
            }
        }

        return result;
    }
}
