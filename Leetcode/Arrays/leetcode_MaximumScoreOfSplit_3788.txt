class Solution {
    public long maximumScore(int[] nums) {
        int n = nums.length;
        long sum = 0;
        long min = nums[n-1];

        for(int i=0; i<n-1; i++) {
            sum += nums[i];
        }

        long result = Long.MIN_VALUE;
        for(int i=n-2; i>=0; i--) {
            result = Math.max(result, sum - min);
            min = Math.min(min, nums[i]);
            sum -= nums[i];
        }

        return result;
    }
}
