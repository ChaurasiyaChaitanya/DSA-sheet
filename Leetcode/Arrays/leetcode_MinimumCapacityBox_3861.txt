class Solution {
    public int minimumIndex(int[] capacity, int itemSize) {
        int minCap = Integer.MAX_VALUE;
        int result = -1;

        for(int i=0; i<capacity.length; i++) {
            if(capacity[i] >= itemSize && minCap > capacity[i]) {
                minCap = capacity[i];
                result = i;
            }
        }

        return result;
    }
}
