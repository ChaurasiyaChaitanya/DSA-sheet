class Solution {
    public int sortPermutation(int[] nums) {
        int n = nums.length;
        int power = 1;

        while(power < n) {
            power <<= 1;
        }

        int maxK = power-1;
        int k = maxK;

        for(int i=0; i<n; i++) {
            if(nums[i] != i) {
                k &= nums[i];
            }
        }

        return k == maxK ? 0 : k;
    }
}
