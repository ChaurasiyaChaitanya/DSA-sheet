class Solution {
    public int minOperations(int[] nums) {
        // take subarray of all elements
        // if two numbers are not equal -> 1
        // if all equal -> 0

        for(int i=0; i<nums.length-1; i++) {
            if(nums[i] != nums[i+1])
                return 1;
        }

        return 0;
    }
}
