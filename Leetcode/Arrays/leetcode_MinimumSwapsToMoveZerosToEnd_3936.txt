class Solution {
    public int minimumSwaps(int[] nums) {
        int n = nums.length;
        int zeros = 0;

        for(int num : nums) {
            if(num == 0) {
                zeros++;
            }
        }

        int result = 0;
        for(int i=0; i<n-zeros; i++) {
            if(nums[i] == 0) {
                result++;
            }
        }

        return result;
    }
}
