class Solution {
    public int minOperations(int n) {
        int mid = n; // middle value
        int result = 0;

        // Only need to balance left half to the target
        for (int i=0; i<n/2; i++) {
            int val = 2 * i + 1;  // arr[i]
            result += mid - val;
        }

        return result;
    }
}
