class Solution {
    public int[] countOppositeParity(int[] nums) {
        int n = nums.length;
        int even = 0, odd = 0;

        int[] result = new int[n];

        for(int i=n-1; i>=0; i--) {
            if(nums[i] % 2 == 1) {
                result[i] = even;
                odd++;
            } else {
                result[i] = odd;
                even++;
            }
        }

        return result;
    }
}
