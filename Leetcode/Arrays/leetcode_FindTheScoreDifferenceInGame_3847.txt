class Solution {
    public int scoreDifference(int[] nums) {
        int p1 = 0, p2 = 0;
        boolean turn = true;

        for(int i=0; i<nums.length; i++) {
            if(nums[i] % 2 == 1) {
                turn = !turn;
            }
            if(i % 6 == 5) {
                turn = !turn;
            }
            if(turn) {
                p1 += nums[i];
            } else {
                p2 += nums[i];
            }
        }

        return p1 - p2;
    }
}
