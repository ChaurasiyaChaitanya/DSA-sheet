class Solution {
    public int findNonMinOrMax(int[] nums) {
        int minVal = Integer.MAX_VALUE, maxVal = Integer.MIN_VALUE;

        for(int num : nums) {
            minVal = Math.min(minVal, num);
            maxVal = Math.max(maxVal, num);
        }

        for(int num : nums) {
            if(num > minVal && num < maxVal) {
                return num;
            }
        }

        return -1;
    }
}
