class Solution {
    public List<Integer> toggleLightBulbs(List<Integer> bulbs) {
        boolean[] bulb = new boolean[101];

        for(int b : bulbs) {
            bulb[b] = !bulb[b];
        }

        List<Integer> result = new ArrayList<>();
        for(int i=1; i<101; i++) {
            if(bulb[i]) {
                result.add(i);
            }
        }

        return result;
    }
}
