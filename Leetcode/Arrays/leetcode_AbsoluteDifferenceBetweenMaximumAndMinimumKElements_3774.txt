class Solution {
    public int absDifference(int[] nums, int k) {
        int n = nums.length;
        Arrays.sort(nums);

        int result = 0;
        for(int i=0, j=n-1; i<k; i++, j--) {
            result += nums[j] - nums[i];
        }
        
        return result;

        // int minSum = 0, maxSum = 0;

        // return maxSum - minSum;
    }
}
