class Solution {
    public List<Integer> findMissingElements(int[] nums) {
        int[] arr = new int[101];

        int minEle = Integer.MAX_VALUE;
        int maxEle = 0;

        for(int num : nums) {
            arr[num] = 1;
            minEle = Math.min(minEle, num);
            maxEle = Math.max(maxEle, num);
        }

        List<Integer> result = new ArrayList<>();

        for(int num=minEle+1; num<maxEle; num++) {
            if(arr[num] == 0) {
                result.add(num);
            }
        }

        return result;
    }
}
