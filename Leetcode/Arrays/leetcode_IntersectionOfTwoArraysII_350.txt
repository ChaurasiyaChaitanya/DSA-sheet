class Solution {
    public int[] intersect(int[] nums1, int[] nums2) {
        int[] avail = new int[1001];

        for(int num : nums1) {
            avail[num]++;
        }

        List<Integer> result = new ArrayList<>();
        for(int num : nums2) {
            if(avail[num] > 0) {
                result.add(num);
                avail[num]--;
            }
        }

        return result.stream().mapToInt(e -> e).toArray();
    }
}
