class Solution {
    public int centeredSubarrays(int[] nums) {
        int n = nums.length;
        int result = 0;

        for(int i=0; i<n; i++) {
            Set<Integer> set = new HashSet<>();
            int sum = 0;
            for(int j=i; j<n; j++) {
                set.add(nums[j]);
                sum += nums[j];
                if(set.contains(sum)) {
                    result++;
                }
            }
        }

        return result;
    }
}
