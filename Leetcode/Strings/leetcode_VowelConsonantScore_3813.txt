class Solution {
    public int vowelConsonantScore(String s) {
        int v = 0, c = 0;

        for(char ch : s.toCharArray()) {
            if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
                v++;
            // } else if(!Character.isDigit(ch) && ch != ' ') {
            } else if(ch >= 'b' && ch <= 'z') {
                c++;
            }
        }

        return c == 0 ? 0 : (int) Math.floor(v/c);
    }
}
