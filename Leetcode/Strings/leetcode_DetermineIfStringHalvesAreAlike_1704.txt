class Solution {
    public boolean halvesAreAlike(String s) {
        Set<Character> set = Set.of('a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U');
        
        int mid = s.length() / 2, result = 0;

        for(int i=0; i<mid; i++) {
            char a = s.charAt(i), b = s.charAt(mid + i);
            if(set.contains(a)) {
                result++;
            }
            if(set.contains(b)) {
                result--;
            }
        }

        return result == 0;
    }
}
