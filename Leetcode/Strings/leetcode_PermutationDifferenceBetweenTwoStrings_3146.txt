class Solution {
    public int findPermutationDifference(String s, String t) {
        int[] pos = new int[26];

        for(int i=0; i<s.length(); i++) {
            pos[s.charAt(i) - 'a'] += i;
            pos[t.charAt(i) - 'a'] -= i;
        }

        int result = 0;
        for(int diff : pos) {
            if(diff != 0) {
                result += Math.abs(diff);
            }
        }

        return result;
    }
}
