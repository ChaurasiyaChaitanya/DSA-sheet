class Solution {
    public int countKeyChanges(String s) {
        int result = 0;

        char ch = Character.toLowerCase(s.charAt(0));

        for(int i=1; i<s.length(); i++) {
            char curr = Character.toLowerCase(s.charAt(i));
            if(curr != ch) {
                result++;
                ch = curr;
            }
        }

        return result;
    }
}
