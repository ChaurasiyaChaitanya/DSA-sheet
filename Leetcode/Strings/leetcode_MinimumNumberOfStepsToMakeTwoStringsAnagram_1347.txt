class Solution {
    public int minSteps(String s, String t) {
        int[] freq = new int[26];

        for(int i=0; i<s.length(); i++) {
            // t have more charcters
            freq[t.charAt(i) - 'a']++;
            freq[s.charAt(i) - 'a']--;
        }

        int result = 0;
        // Need to swap one so ultimately other -ve freq chararter becomes equal
        for(int count : freq) {
            result += Math.max(0, count);
        }

        return result;
    }
}
