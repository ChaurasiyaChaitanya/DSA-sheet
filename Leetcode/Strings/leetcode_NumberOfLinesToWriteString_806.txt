class Solution {
    public int[] numberOfLines(int[] widths, String s) {
        int lines = 1, pixels = 0;

        for(int i=0; i<s.length(); i++) {
            int c = widths[s.charAt(i) - 'a'];
            if(pixels + c > 100) {
                pixels = c;
                lines++;
            } else {
                pixels += c;
            }
        }

        return new int[]{lines, pixels};
    }
}
