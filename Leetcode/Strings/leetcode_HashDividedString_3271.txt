class Solution {
    public String stringHash(String s, int k) {
        int n = s.length();
        StringBuilder sb = new StringBuilder();

        for(int i=0; i<n; i+=k) {
            int sum = 0;
            for(int j=i; j<i+k; j++) {
                sum += s.charAt(j) - 'a';
            }
            sb.append((char) ('a' + (sum % 26)));
        }

        return sb.toString();
    }
}
