class Solution {
    public boolean scoreBalance(String s) {
        int n = s.length();
        int totalSum = 0;

        for(char c : s.toCharArray()) {
            totalSum += (c - 'a' + 1);
        }

        int currSum = 0;
        for(int i=0; i<n-1; i++) {
            currSum += (s.charAt(i) - 'a' + 1);
            if(2 * currSum == totalSum) {
                return true;
            }
        }

        return false;
    }
}
