class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int[] pos = new int[26];
        Arrays.fill(pos, -1);

        int result = -1;

        for(int i=0; i<s.length(); i++) {
            int c = s.charAt(i) - 'a';
            if(pos[c] == -1) {
                pos[c] = i;   // store first occurrence only
            } else {
                result = Math.max(result, i - pos[c] - 1);
            }
        }

        return result;
    }
}
