class Solution {
    public int minLengthAfterRemovals(String s) {
        int count1 = 0, count2 = 0;

        for(char c : s.toCharArray()) {
            if(c == 'a') {
                count1++; 
            } else {
                count2++;
            }
        }

        return Math.abs(count1 - count2);
    }
}
