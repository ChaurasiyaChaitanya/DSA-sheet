class Solution {
    public String makeSmallestPalindrome(String s) {
        int n = s.length();
        int i = 0, j = n - 1;

        char[] ch = s.toCharArray();

        while(i < j) {
            ch[i] = (char) Math.min(ch[i], ch[j]);
            ch[j]= ch[i];
            i++;
            j--;
        }

        return new String(ch);
    }
}
