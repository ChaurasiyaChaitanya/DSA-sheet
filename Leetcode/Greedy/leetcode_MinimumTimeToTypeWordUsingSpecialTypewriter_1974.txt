class Solution {
    public int minTimeToType(String word) {
        int result = 0;

        result += Math.min(Math.abs(word.charAt(0) - 'a'), Math.abs(26 - (word.charAt(0) - 'a'))) + 1;

        for(int i=1; i<word.length(); i++) {
            int clock = Math.abs(word.charAt(i) - word.charAt(i-1));
            result += Math.min(clock, 26 - clock) + 1; 
        }

        return result;
    }
}
