class Solution {
    public long maxSum(int[] nums, int k, int mul) {
        int n = nums.length;
        Arrays.sort(nums);

        long result = 0;
        for(int i=n-1; i>=n-k; i--) {
            if(mul > 1) {
                result += 1L * nums[i] * mul;
                mul--;
            } else {
                result += nums[i];
            }
        }

        return result;
    }
}
