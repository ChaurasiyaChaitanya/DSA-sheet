class Solution {
    public int maximumBags(int[] capacity, int[] rocks, int additionalRocks) {
        
        for(int i=0; i<rocks.length; i++) {
            rocks[i] = capacity[i] - rocks[i];
        }

        Arrays.sort(rocks);

        int result = 0;
        for(int i=0; i<rocks.length; i++) {
            if(additionalRocks >= rocks[i]) {
                additionalRocks -= rocks[i];
                result++;
            } else {
                break;
            }
        }

        return result;
    }
}
