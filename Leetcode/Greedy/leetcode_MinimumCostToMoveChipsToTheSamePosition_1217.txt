class Solution {
    public int minCostToMoveChips(int[] position) {
        int result = 0;
        int evenCount = 0, oddCount = 0;

        // check min between even and odd position chips
        for(int pos : position) {
            if(pos % 2 == 0) evenCount++;
            else oddCount++;
        }

        return Math.min(evenCount, oddCount);
    }
}
