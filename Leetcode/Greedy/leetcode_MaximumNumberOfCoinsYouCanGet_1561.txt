class Solution {
    public int maxCoins(int[] piles) {
        Arrays.sort(piles);

        int i = piles.length - 2, result = 0;
        int k = piles.length / 3;

        while(k > 0) {
            result += piles[i];
            i -= 2;
            k--;
        }
        
        return result;
    }
}
