class Solution {
    public int maximizeSum(int[] nums, int k) {
        int maxNum = 0, result = 0;

        // Greedily take max element
        for(int num : nums) {
            maxNum = Math.max(maxNum, num);
        }

        while(k > 0) {
            result += maxNum;
            maxNum += 1;
            k--;
        }

        return result;
    }
}
