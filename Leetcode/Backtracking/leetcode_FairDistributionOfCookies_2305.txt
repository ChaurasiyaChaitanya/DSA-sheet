class Solution {
    int result = Integer.MAX_VALUE;

    public int distributeCookies(int[] cookies, int k) {
        int[] child = new int[k];

        solve(0, cookies, child, k);

        return result;
    }

    void solve(int idx, int[] cookies, int[] child, int k) {
        if(idx == cookies.length) {
            int maxCookies = 0;
            for(int val : child) {
                maxCookies = Math.max(maxCookies, val);
            }
            result = Math.min(result, maxCookies);
            return;
        }

        for(int i=0; i<k; i++) {
            child[i] += cookies[idx];
            solve(idx+1, cookies, child, k);
            child[i] -= cookies[idx];
        }
    }
}
