class Solution {
    List<String> result = new ArrayList<>();

    public List<String> letterCasePermutation(String s) {
        solve(0, s, new StringBuilder());

        return result;
    }

    private void solve(int idx, String s, StringBuilder sb) {
        if(idx == s.length()) {
            result.add(sb.toString());
            return;
        }

        char ch = s.charAt(idx);

        if(Character.isDigit(ch)) {
            solve(idx+1, s, sb.append(ch));
            sb.deleteCharAt(sb.length() - 1);
        } else {
            solve(idx+1, s, sb.append(Character.toLowerCase(ch)));
            sb.deleteCharAt(sb.length() - 1);

            solve(idx+1, s, sb.append(Character.toUpperCase(ch)));
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
