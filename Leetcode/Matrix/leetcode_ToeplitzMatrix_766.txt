class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;

        // i = 0, j = 0 - corner has unique elements so starts with 1st row and col
        for(int i=1; i<m; i++) {
            for(int j=1; j<n; j++) {
                if(matrix[i][j] != matrix[i-1][j-1]) { // Prev Diagonal Cell (i-1, j-1)
                    return false;
                }
            }
        }

        return true;
    }
}
